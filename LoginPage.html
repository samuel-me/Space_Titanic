Login Page





from pyodide.ffi import create_proxy def login(event): # Get the username and password input values username = Element("username").element.value password = Element("password").element.value if username and password: # Display a welcome message using the username Element("welcome-message").element.innerText = f"Welcome, {username}!" else: Element("welcome-message").element.innerText = "Please enter both username and password." # Bind the button click to the login function login_btn = Element("login-btn") login_btn.element.addEventListener("click", create_proxy(login))
